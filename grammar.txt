S -> CODE
CODE -> CODE NEW_CODE | NEWLINE CODE | CODE NEWLINE | IF_LOOP
NEW_CODE -> NEWLINE CODE
NEWLINE -> @

//if thing
IF_LOOP -> IF_HEAD | IF_HEAD CODE
IF_HEAD -> IF IF_HEAD_CONDITION_NEWLINE
IF_HEAD_CONDITION -> CONDITION COLON
IF_HEAD_CONDITION_NEWLINE -> IF_HEAD_CONDITION NEWLINE

//conditional operations
CONDITION -> OPERAND LOGIC_OPERAND_OPERATOR | SINGLE_LOGIC_OPERATOR OPERAND
LOGIC_OPERAND_OPERATOR -> LOGIC_OPERATOR OPERAND
LOGIC_OPERATOR -> < | > | >= | <= | != | == | and | or
SINGLE_LOGIC_OPERATOR -> not

//misc
COLON -> :
IF -> a
OPERAND -> /
